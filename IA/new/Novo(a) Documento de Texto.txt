<?php
use PHPMailer\PHPMailer\PHPMailer;
use PHPMailer\PHPMailer\Exception;

require __DIR__ . '/vendor/autoload.php';

$mail = new PHPMailer(true);

try {
 $mail->isSMTP();
 $mail->Host = 'smtp.hostinger.com';
 $mail->SMTPAuth = true;
 $mail->Username = 'seu-email@aceca.com.br';
 $mail->Password = 'SUA_SENHA';
 $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
 $mail->Port = 587;

 $mail->setFrom('seu-email@aceca.com.br', 'Seu Nome');
 $mail->addAddress('destino@exemplo.com');

 $mail->isHTML(true);
 $mail->Subject = 'Teste PHPMailer';
 $mail->Body = 'Funcionou!';

 $mail->send();
 echo 'Email enviado';
} catch (Exception $e) {
 echo 'Falha ao enviar: ' . $mail->ErrorInfo;
}
